345  template<class Generator> void iterate(Generator mdp)
346  {
347  //***********************************************************************
348  // Create stack to hold the path visited on this iteration.
349  // Initially this holds only the current node.
350  //***********************************************************************
351  std::stack<UCTreeNode*> visited;
352  UCTreeNode* pCur = this;
353  visited.push(this);
354 
355  //***********************************************************************
356  // Create stack to hold immediate rewards generated as we transverse
357  // the tree. Initially, this zero as a place holder for the reward
358  // received for the root node. This has no real effect on the result,
359  // but simplifies the backup algorithm applied later on.
360  //***********************************************************************
361  std::stack<double> rewards;
362  rewards.push(0.0);
363 
364  //***********************************************************************
365  // Transverse the highest value path from the current node, until
366  // we hit a leaf node. We also record rewards for each action as we go
367  // along.
368  //***********************************************************************
369  int action = 0; // next selected action
370  double curReward = 0.0; // immediate reward for last action
371  while (!pCur->isLeaf())
372  {
373  action = pCur->selectAction();
374  pCur = pCur->vpChildren_i[action];
375  visited.push(pCur);
376  curReward = mdp(action);
377  rewards.push(curReward);
378  }
379 
380  //***********************************************************************
381  // Expand the current (leaf) node by depth 1, and select its best child.
382  //***********************************************************************
383  pCur->expand();
384  action = pCur->selectAction();
385  pCur = pCur->vpChildren_i[action];
386  visited.push(pCur);
387  curReward = mdp(action);
388  rewards.push(curReward);
389 
390  //***********************************************************************
391  // Estimate the value of the new leaf node using the rollout policy
392  //***********************************************************************
393  double value = rollOut(mdp);
394 
395  //***********************************************************************
396  // Update the statistics for each node along the path using the
397  // discounted value.
398  //***********************************************************************
399  while(!visited.empty())
400  {
401  assert(visited.size()==rewards.size()); // should always be true
402  value = rewards.top() + gamma_i*value; // update the total value
403  pCur = visited.top(); // get the current node in the path
404  pCur->updateStats(value); // update statistics
405  visited.pop(); // remove the current node from the stack
406  rewards.pop();
407  }
408 
409  } // iterate
